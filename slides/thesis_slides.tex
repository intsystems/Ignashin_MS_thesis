\begin{frame}{Положения, выносимые на защиту}
    \footnotesize
    \begin{enumerate}
        \item Рассмотрены модификации Frank--Wolfe, сохраняющие проекционно-свободную итерацию и улучшающие практическую сходимость за счет выбора направления, блочного LMO и распределенной архитектуры.
        \item Для модели Бекмана предложены NFW как обобщение CFW/BFW на $N$ прошлых направлений и WFFW как модификация FFW с экспоненциальным сглаживанием.
        \item CFW и NFW перенесены на ML-задачи без явного гессиана: кривизна аппроксимируется разностями градиентов.
        \item Для CFW доказана оценка по минимальному зазору Frank--Wolfe того же порядка, что у базового FW.
        \item Предложен SOFW как частный случай BCFW для транспортной задачи с блоками по OD-парам; доказана оценка ожидаемой сходимости для одиночного и батчевого выбора блоков.
        \item Предложен DBCFW: децентрализованный блочно-координатный FW с консенсусом и отслеживанием градиента; получена оценка сходимости порядка $O(1/t)$.
    \end{enumerate}
\end{frame}

% 3. Общая постановка
% =====================================================================
\begin{frame}{Постановка задачи}
    \small
    \vspace{-1mm}
    Рассматривается гладкая выпуклая задача на компактном выпуклом множестве:
    \[
        f(x) \to \min_{x\in\cX}.
    \]
    Метод Frank--Wolfe заменяет проекцию линейным оракулом:
    \[
        s_k^{FW}\in \argmin_{s\in\cX}\inner{\nabla f(x_k)}{s},
        \qquad
        d_k^{FW}=s_k^{FW}-x_k,
    \]
    \[
        x_{k+1}=x_k+\gamma_k d_k^{FW},
        \qquad
        \gamma_k\in[0,1].
    \]
    \begin{block}{Основной вопрос}
        Как сохранить проекционно-свободную структуру FW и ускорить практическую сходимость за счет направления, стохастического LMO и распределенной архитектуры?
    \end{block}
    \vspace{-1mm}
    Качество итерации измеряется зазором
    \[
        g_k=\inner{\nabla f(x_k)}{x_k-s_k^{FW}}.
    \]
\end{frame}

% =====================================================================
% 4. Модель Бекмана и линейный оракул
% =====================================================================
\begin{frame}{Транспортная модель Бекмана}
    \small
    \vspace{-2mm}
    Для дорожной сети поток по ребрам $f=\Theta x$ находится из задачи
    \[
        \Psi(f)=\sum_{e\in\mathcal{E}}\int_0^{f_e}\tau_e(z)\,dz
        \to
        \min_{f=\Theta x,\;x\in X},
        \qquad
        \nabla\Psi(f)=\tau(f).
    \]
    Допустимое множество имеет блочную структуру по OD-парам:
    \[
        X=\prod_{w\in OD}X^w,\qquad
        X^w=\left\{x^w:\sum_{p\in\cP_w}x_p=d_w,\;x_p\ge 0\right\}.
    \]
    Линейный оракул сводится к поиску кратчайших путей для OD-пар.
    \vspace{-1mm}

% 5. Модификации направления
% =====================================================================
\begin{frame}{Модификации направления Frank--Wolfe}
    \footnotesize
    Классический FW использует только текущий оракул $s_k^{FW}$. Модификации добавляют память о предыдущих направлениях.
    \vspace{1.5mm}

    \textbf{Conjugate Frank--Wolfe}
    \[
        s_k=\alpha s_k^{FW}+(1-\alpha)s_{k-1},
        \qquad
        (d^{k-1})^\top \nabla^2\Psi(f^k)d^k=0.
    \]
    \vspace{-2mm}

    \textbf{N-conjugate Frank--Wolfe}
    \[
        s_k=\alpha_0s_k^{FW}+\sum_{j=1}^{N}\alpha_js_{k-j},
        \qquad
        (d^k)^\top\nabla^2\Psi(f^k)d^{k-j}=0.
    \]
    \vspace{-2mm}

    \textbf{Fukushima и Weighted Fukushima FW}
    \[
        s_k=\alpha_0s_k^{FW}+(1-\alpha_0)\sum_{j=1}^{N} w_js_{k-j},
        \qquad
        w_j\propto (1-\eta)\eta^{j-1}.
    \]
    В модели Бекмана $\nabla^2\Psi(f)$ диагонален, поэтому коэффициенты направлений вычисляются дешево.
\end{frame}

% 7. CFW в машинном обучении
% =====================================================================
\begin{frame}{CFW в задачах машинного обучения}
    В ML-задачах явный гессиан часто слишком дорогой. Поэтому произведение гессиана на предыдущее направление заменяется разностью градиентов:
    \[
        \nabla^2 f(x_k)(s_{k-1}-x_{k-1})
        \approx
        \frac{\nabla f(x_k)-\nabla f(x_{k-1})}{\gamma_{k-1}}.
    \]
    После сокращения $\gamma_{k-1}$ коэффициент CFW вычисляется как
    \[
        \alpha=
        \frac{(x_k-s_{k-1})^\top(\nabla f(x_k)-\nabla f(x_{k-1}))}
        {(s_k^{FW}-s_{k-1})^\top(\nabla f(x_k)-\nabla f(x_{k-1}))}.
    \]
    \begin{block}{Экспериментальная постановка}
        Логистическая регрессия на Mushrooms и MNIST:
        \[
            \norm{w}_1\le R,
            \qquad
            f(w)=\operatorname{logloss}(w)+\frac{\lambda}{2}\norm{w}_2^2.
        \]
    \end{block}
\end{frame}

% =====================================================================
% 8. Теорема CFW и эксперименты ML
% =====================================================================
\begin{frame}{Теорема 1. Сходимость CFW}
    \footnotesize
    \vspace{-2mm}
    \begin{theorem_rus}[Оценка по зазору Frank--Wolfe]
        Пусть $\cX\subset\R^d$ выпукло и компактно, $f$ выпукла, непрерывно дифференцируема и имеет $L$-липшицев градиент на $\cX$. Тогда для CFW с параметром $\alpha_0\in(0,1]$
        \[
            g_N^\ast=\Min_{1\le k\le N}g_k
            \le
            \frac{\max\{2h_0,LR^2\}}{\alpha_0\sqrt{N+1}},
            \quad
            h_0=f(x_0)-f(x^\ast).
        \]
    \end{theorem_rus}
    Оценка сохраняет порядок классического FW по минимальному зазору; отличие входит через контролируемый параметр $\alpha_0$.
    \vspace{-1mm}
    \begin{figure}
        \centering
        \includegraphics[width=0.72\linewidth]{cfw_ml/Experiment_5_fig_12.pdf}
    \end{figure}
\end{frame}

% =====================================================================
% 9. SOFW
% =====================================================================
\begin{frame}{SOFW как BCFW в транспортной задаче}
    \vspace{-2mm}
    Предложен SOFW как частный случай BCFW для транспортной задачи с блочной структурой по OD-парам. Полный LMO требует запускать кратчайшие пути для всех блоков, а SOFW обновляет только случайный блок:
    \[
        w_k\sim \operatorname{Unif}(OD),
        \qquad
        s_k^{[w_k]}\in
        \argmin_{u\in X^{w_k}}\inner{\nabla\Psi(F(x_k))}{F(u^{[w_k]})}.
    \]
    \[
        x_{k+1}=x_k+\gamma_k(s_k^{[w_k]}-x_k^{[w_k]}).
    \]
    Батчевый вариант выбирает подмножество $\cI_k$ размера $m$:
    \[
        d_k=\frac{1}{m}\sum_{w\in\cI_k}(s_k^{[w]}-x_k^{[w]}).
    \]
    \begin{block}{Компромисс}
        Полный FW дает более точное направление, но дорогую итерацию. SOFW как BCFW для транспортной сети уменьшает стоимость итерации и выигрывает по времени на крупных графах.
    \end{block}
\end{frame}

% =====================================================================
% 10. Теорема SOFW и результаты
% =====================================================================
\begin{frame}{Теорема 2. Оценка SOFW}
    \footnotesize
    \vspace{-2mm}
    \begin{theorem_rus}[Сходимость батчевого SOFW]
        Пусть $\{x_k\}$ порождается батчевым SOFW, $|OD|$ -- число блоков, а
        \[
            \gamma_k=\frac{2|OD|}{k+2|OD|}.
        \]
        Тогда
        \[
            \E[\Psi(F(x_k))]-\Psi(F(x^\ast))
            \le
            \frac{2|OD|}{k+2|OD|}
            \left[
                C_\Psi^\otimes+\Psi(F(x_0))-\Psi(F(x^\ast))
            \right].
        \]
        Для достижения $\varepsilon$-точности в ожидании достаточно
        $\mathcal{O}(|OD|/\varepsilon)$ итераций.
    \end{theorem_rus}
    \vspace{-2mm}

        \centering
        \includegraphics[width=0.54\linewidth]{scfw/Philadelphia.pdf}
    \end{figure}
\end{frame}

% 11. DBCFW
% =====================================================================
\begin{frame}{DBCFW: блочный FW в распределенной оптимизации}
    \footnotesize
    \vspace{-1mm}
    Распределенная задача:
    \[
        f(x)=\frac{1}{N}\sum_{i=1}^{N} f_i(x)\to\min_{x\in D_1\times\dots\times D_n}.
    \]
    Агент $i$ смешивает точки и оценки градиента:
    \[
        \tilde x_t^i=\sum_{j=1}^{N}(W_t)_{ij}x_t^j,
        \qquad
        g_t^i=\tilde g_{t-1}^i+\nabla f_i(\tilde x_t^i)-\nabla f_i(\tilde x_{t-1}^i),
    \]
    \[
        \tilde g_t^i=\sum_{j=1}^{N}(W_t)_{ij}g_t^j.
    \]
    Затем выбирается батч блоков $I_t^i$ и суженное множество
    \[
        D_t^i=
        \left(\prod_{k\in I_t^i}D_k\right)
        \times
        \left(\prod_{k\notin I_t^i}\{\tilde x_t^i[k]\}\right),
    \]
    Блочный LMO:
    \[
        v_t^i\in\argmin_{v\in D_t^i}\inner{\tilde g_t^i}{v},
        \qquad
        x_{t+1}^i=(1-\gamma_t)\tilde x_t^i+\gamma_t v_t^i.
    \]
\end{frame}

% =====================================================================
% 12. Теорема DBCFW и карта методов
% =====================================================================
\begin{frame}{Теорема 3. Сходимость DBCFW}
    \scriptsize
    \vspace{-2mm}
    \begin{theorem_rus}[Оценка для среднего итерата]
        Пусть выполнены условия смешивания, гладкости и ограниченности области. Пусть число блоков равно $n$, размер батча $1\le B\le n$,
        \[
            \tau=\frac{2n}{B},
            \qquad
            \gamma_t=\frac{2n}{B(t+\tau)}.
        \]
        Если консенсусная ошибка и ошибка градиентного отслеживания убывают как $O(1/t)$, то
        \[
            \E[f(x_{\mathrm{avg},t})-f(x^\ast)]
            \le
            \frac{\widetilde A_\tau}{t+\tau}.
        \]
    \end{theorem_rus}
    \vspace{-2mm}

    \centering

\end{frame}
